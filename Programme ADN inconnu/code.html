function typeContext(str)
        {
            // make conversation
            $('#context').typed({
                strings: str,
                typeSpeed: 5,
                backSpeed: 0,
                backDelay: 10,
                showCursor: true,
                cursorChar: "|"
            });
        }
        
        function countContext(str)
        {
            return str.length;
        }
